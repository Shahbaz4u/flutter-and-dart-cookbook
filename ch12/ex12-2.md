# 12.2 Referencing an Image

# Example

```dart
import 'package:flutter/material.dart';

void main() {
 runApp(const MyApp());
}
class MyApp extends StatelessWidget {
 const MyApp({Key? key}) : super(key: key);
 @override
    Widget build(BuildContext context) {
        const title = 'Image Asset demo';
        return MaterialApp(
            title: title,
            home: Scaffold(
            appBar: AppBar(
                title: const Text(title),
             ),
             body: const MyImageAssetWidget(),
            ),
         );
    }
}
class MyImageAssetWidget extends StatelessWidget {
 const MyImageAssetWidget({Key? key}) : super(key: key);
 @override
 Widget build(BuildContext context) {
    return const Center(
        child: Image.asset('images/green-tile.png'),
        );
     }
}

```
